react에 style 적용하는 방법

style.css 파일 생성

button {
  color: #fff;
  background-color: tomato;
}

index.js

...

import "./style.css"

...

직접 style 객체 입력

const Button = ({ text }) => {
  <button style={{
  color: "#fff";
  backgroundColor: "tomato";
  }}
  >{text}</button>
}

.module.css 파일 만들기

Button.module.css

.btn {
  background-color: tomato;
  color: white;
}

Buton.jsx

import styles from "./Button.module.css";

const Button = ({ text }) => {
  <button className={styles.btn}>{text}</button>;
};
